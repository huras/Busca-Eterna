public $myAttachments;

use Illuminate\Bus\Queueable;

[ ] ================= Na função build, vc monta o email com as coias que quer e tudo mais, é aqui q atachamos

public function build()
    {
        $message = $this->view('email.associeseEmail')->subject('Gremig Associados');
        foreach($this->myAttachments as $path){
	    $thePath = storage_path(str_replace ('/','\\',$path));//Muda a orientação original das barras
            $thePath = str_replace('\storage','\public\storage',$thePath);//Acrescenta o diretorio
            $message->attach($thePath);
        }
        return $message;
    }
